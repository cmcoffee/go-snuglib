
# cfg
    import "github.com/cmcoffee/go-cfg"

Package 'cfg' provides functions for reading and writing configuration files and their coresponding string values.


	Ignores '#', ';' as comment lines, ','s denote multiple values.
	
	# Example config file.
	[section]
	key = value
	key2 = value1, value2
	key3 = value1,
	       value2,
	       value3
	
	[section2]
	key = value1,
	      value2,
	      value3
